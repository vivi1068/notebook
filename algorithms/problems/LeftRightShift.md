### 位移
```python
# 字符
ss = "a"

# 转成数字
ss_int = ord(ss)                            # 97

# 转成字符
ss_back = chr(ss_int)                       # 'a'

# 转成二进制
int_to_binary = bin(ss_int)[2:].zfill(8)    # '01100001'

# 左移(变大): 整体左移1位
int_left_shift = ss_int << 1                # '11000010'        

# 右移(变小): 整体右移1位
int_right_shift = ss_int >> 1               # '00110000'

```

### 位移口诀
```python
# 左移(变大): 左移1位 == 乘以1次2
# 左移(变大): 左移2位 == 乘以2次2
# 左移(变大): 左移3位 == 乘以3次2
# 左移(变大): 左移4位 == 乘以4次2
# 左移(变大): 左移5位 == 乘以5次2
ss = 97
leftshift_byone = ss << 1       # 97 * 2 = 194                      # '00000000 11000010'
leftshift_bytwo = ss << 2       # 97 * 2 * 2 = 388                  # '00000001 10000100'
leftshift_bythr = ss << 3       # 97 * 2 * 2 * 2 = 776              # '00000011 00001000'
leftshift_byfou = ss << 4       # 97 * 2 * 2 * 2 * 2 = 1552         # '00000110 00010000'
leftshift_byfiv = ss << 5       # 97 * 2 * 2 * 2 * 2 * 2 = 3104     # '00001100 00100000'


# 右移(变小): 右移1位 == 除以1次2
# 右移(变小): 右移2位 == 除以2次2
# 右移(变小): 右移3位 == 除以3次2
# 右移(变小): 右移4位 == 除以4次2
# 右移(变小): 右移5位 == 除以5次2
rightshift_byone = ss >> 1      # 97 // 2 = 48                      # '00000000 00110000'
rightshift_bytwo = ss >> 2      # 97 // 2 // 2 = 24                 # '00000000 00011000'
rightshift_bythr = ss >> 3      # 97 // 2 // 2 // 2 = 12            # '00000000 00001100'
rightshift_byfou = ss >> 4      # 97 // 2 // 2 // 2 // 2 = 6        # '00000000 00000110'
rightshift_byfiv = ss >> 5      # 97 // 2 // 2 // 2 // 2 // 2 = 3   # '00000000 00000011'
```
